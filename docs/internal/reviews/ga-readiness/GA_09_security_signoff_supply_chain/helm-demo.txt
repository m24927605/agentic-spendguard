            - { name: SPENDGUARD_SIDECAR_UDS_PATH,     value: "/var/run/spendguard/adapter.sock" }

            - { name: SPENDGUARD_SIDECAR_ENDPOINT_CATALOG_MANIFEST_URL, value: "https://endpoint-catalog.spendguard:8443/v1/catalog/manifest" }
            - { name: SPENDGUARD_SIDECAR_FENCING_INITIAL_EPOCH,         value: "1" }

          volumeMounts:

            - { name: bundles,             mountPath: /var/lib/spendguard/bundles, readOnly: true }

            items:

apiVersion: batch/v1
kind: Job
metadata:
  name: spendguard-spendguard-migrate
  labels:
    app.kubernetes.io/name: spendguard
    app.kubernetes.io/instance: spendguard
    app.kubernetes.io/version: "0.1.0-alpha.3"
    app.kubernetes.io/managed-by: Helm
    helm.sh/chart: spendguard-0.1.0-alpha.3
    app.kubernetes.io/component: migrations
  annotations:
    "helm.sh/hook": post-install,post-upgrade
    "helm.sh/hook-weight": "-5"
    "helm.sh/hook-delete-policy": before-hook-creation,hook-succeeded
spec:
  backoffLimit: 3
  template:
    metadata:
      labels:
        app.kubernetes.io/name: spendguard
        app.kubernetes.io/instance: spendguard
        app.kubernetes.io/version: "0.1.0-alpha.3"
        app.kubernetes.io/managed-by: Helm
        helm.sh/chart: spendguard-0.1.0-alpha.3
        app.kubernetes.io/component: migrations
    spec:
      restartPolicy: OnFailure
      securityContext:
        runAsNonRoot: true
        runAsUser: 65532
        runAsGroup: 65532
        fsGroup: 65532
        seccompProfile:
          type: RuntimeDefault
      containers:
        - name: migrate
          image: postgres:16-alpine
          securityContext:
            readOnlyRootFilesystem: true
            allowPrivilegeEscalation: false
            capabilities:
              drop: ["ALL"]
          command: ["/bin/sh", "-c"]
          args:
            - |
              set -eu
              applied=0
              skipped=0
              # Round-3 fix m6 + round-4 fix B2: detect SLICE_01-incomplete
              # ConfigMap and warn loudly. With the production fail-gate
              # removed (round-4 B2), this is the ONLY signal the operator
              # gets from the chart that the ConfigMap mount is missing
              # the SLICE_01 files. Warning is visible in:
              #   * helm install --wait | --debug
              #   * kubectl logs -l app.kubernetes.io/component=migrations
              #   * The kind-cluster job at .github/workflows/helm-validate.yml
              missing_slice01=""
              for f in 0046_audit_outbox_prediction_columns.sql 0048_tokenizer_versions.sql; do
                  if [ -d /etc/migrations/ledger ] && [ ! -f "/etc/migrations/ledger/$f" ]; then
                      missing_slice01="$missing_slice01 ledger/$f"
                  fi
              done
              for f in 0013_canonical_events_prediction_columns.sql 0015_audit_outcome_quarantine_prediction_columns.sql; do
                  if [ -d /etc/migrations/canonical ] && [ ! -f "/etc/migrations/canonical/$f" ]; then
                      missing_slice01="$missing_slice01 canonical/$f"
                  fi
              done
              if [ -n "$missing_slice01" ]; then
                  echo "[migrate] WARNING: SLICE_01 migration file(s) MISSING from mounted ConfigMaps:$missing_slice01"
                  echo "[migrate] WARNING: this will cause a 'column does not exist'"
                  echo "[migrate] WARNING: runtime error once SLICE_06 producers attempt to write tag-300+ fields."
                  echo "[migrate] WARNING: regenerate ConfigMaps via scripts/helm-validate-kind.sh."
              fi
              for f in $(ls /etc/migrations/ledger/*.sql 2>/dev/null | sort); do
                  echo "[migrate] applying $f to spendguard_ledger..."
                  psql "$PG_LEDGER_URL" -v ON_ERROR_STOP=1 -f "$f"
                  applied=$((applied + 1))
              done
              for f in $(ls /etc/migrations/canonical/*.sql 2>/dev/null | sort); do
                  echo "[migrate] applying $f to spendguard_canonical..."
                  psql "$PG_CANONICAL_URL" -v ON_ERROR_STOP=1 -f "$f"
                  applied=$((applied + 1))
              done
              # R2 M3 (Security F5): control_plane migrations apply against
              # the separate spendguard_control_plane DB. The ConfigMap is
              # OPTIONAL — if absent, no-op. Production helm-validate-kind.sh
              # regenerates all three ConfigMaps in lockstep.
              for f in $(ls /etc/migrations/control-plane/*.sql 2>/dev/null | sort); do
                  echo "[migrate] applying $f to spendguard_control_plane..."
                  psql "$PG_CONTROL_PLANE_URL" -v ON_ERROR_STOP=1 -f "$f"
                  applied=$((applied + 1))
              done
              if [ "$applied" -eq 0 ]; then
                  echo "[migrate] no SQL files found in /etc/migrations/{ledger,canonical,control-plane}/."
                  echo "[migrate] Operator must mount the spendguard-migrations-ledger +"
                  echo "[migrate] spendguard-migrations-canonical +"
                  echo "[migrate] spendguard-migrations-control-plane ConfigMaps OR"
                  echo "[migrate] override migrations.applyMigrationsOnInstall=false + run migrations out-of-band."
                  echo "[migrate] Job exits 0 so helm install can complete."
              else
                  echo "[migrate] PASS — applied ${applied} migration file(s)."
              fi
          env:
            - name: PG_LEDGER_URL
              valueFrom:
                secretKeyRef:
                  name: "spendguard-postgres-urls"
                  key: "ledger-url"
            - name: PG_CANONICAL_URL
              valueFrom:
                secretKeyRef:
                  name: "spendguard-postgres-urls"
                  key: "canonical-url"
            - name: PG_CONTROL_PLANE_URL
              valueFrom:
                secretKeyRef:
                  name: "spendguard-postgres-urls"
                  key: "control-plane-url"
          volumeMounts:
            - name: migrations-ledger
              mountPath: /etc/migrations/ledger
              readOnly: true
            - name: migrations-canonical
              mountPath: /etc/migrations/canonical
              readOnly: true
            - name: migrations-control-plane
              mountPath: /etc/migrations/control-plane
              readOnly: true
          resources:
            limits:
              cpu: 500m
              memory: 256Mi
            requests:
              cpu: 100m
              memory: 128Mi
      volumes:
        - name: migrations-ledger
          configMap:
            name: spendguard-migrations-ledger
            optional: true
        - name: migrations-canonical
          configMap:
            name: spendguard-migrations-canonical
            optional: true
        - name: migrations-control-plane
          configMap:
            name: spendguard-migrations-control-plane
            optional: true
